92      protected List<Box> createMoofMdat(final Movie movie) {
93          List<Box> boxes = new LinkedList<Box>();
94          int maxNumberOfFragments = 0;
95          for (Track track : movie.getTracks()) {
96              int currentLength = intersectionFinder.sampleNumbers(track, movie).length;
97              maxNumberOfFragments = currentLength > maxNumberOfFragments ? currentLength : maxNumberOfFragments;
98          }
99          int sequence = 1;
100         for (int i = 0; i < maxNumberOfFragments; i++) {
101 
102             final List<Track> sizeSortedTracks = new LinkedList<Track>(movie.getTracks());
103             final int j = i;
104             Collections.sort(sizeSortedTracks, new Comparator<Track>() {
105                 public int compare(Track o1, Track o2) {
106                     long[] startSamples1 = intersectionFinder.sampleNumbers(o1, movie);
107                     long startSample1 = startSamples1[j];
108                     // one based sample numbers - the first sample is 1
109                     long endSample1 = j + 1 < startSamples1.length ? startSamples1[j + 1] : o1.getSamples().size() + 1;
110                     long[] startSamples2 = intersectionFinder.sampleNumbers(o2, movie);
111                     long startSample2 = startSamples2[j];
112                     // one based sample numbers - the first sample is 1
113                     long endSample2 = j + 1 < startSamples2.length ? startSamples2[j + 1] : o2.getSamples().size() + 1;
114                     List<ByteBuffer> samples1 = o1.getSamples().subList(l2i(startSample1) - 1, l2i(endSample1) - 1);
115                     List<ByteBuffer> samples2 = o2.getSamples().subList(l2i(startSample2) - 1, l2i(endSample2) - 1);
116                     int size1 = 0;
117                     for (ByteBuffer byteBuffer : samples1) {
118                         size1 += byteBuffer.limit();
119                     }
120                     int size2 = 0;
121                     for (ByteBuffer byteBuffer : samples2) {
122                         size2 += byteBuffer.limit();
123                     }
124                     return size1 - size2;
125                 }
126             });
127 
128             for (Track track : sizeSortedTracks) {
129                 if (getAllowedHandlers().isEmpty() || getAllowedHandlers().contains(track.getHandler())) {
130                     long[] startSamples = intersectionFinder.sampleNumbers(track, movie);
131 
132                     if (i < startSamples.length) {
133                         long startSample = startSamples[i];
134                         // one based sample numbers - the first sample is 1
135                         long endSample = i + 1 < startSamples.length ? startSamples[i + 1] : track.getSamples().size() + 1;
136 
137                         if (startSample == endSample) {
138                             // empty fragment
139                             // just don't add any boxes.
140                         } else {
141                             boxes.add(createMoof(startSample, endSample, track, sequence));
142                             boxes.add(createMdat(startSample, endSample, track, sequence++));
143                         }
144 
145                     } else {
146                         //obvious this track has not that many fragments
147                     }
148                 }
149             }
150 
151 
152         }
153         return boxes;
154     }

159     public IsoFile build(Movie movie) throws IOException {
160         LOG.fine("Creating movie " + movie);
161         IsoFile isoFile = new IsoFile();
162 
163 
164         isoFile.addBox(createFtyp(movie));
165         isoFile.addBox(createMoov(movie));
166 
167         for (Box box : createMoofMdat(movie)) {
168             isoFile.addBox(box);
169         }
170         isoFile.addBox(createMfra(movie, isoFile));
171 
172         return isoFile;
173     }

175     protected Box createMdat(final long startSample, final long endSample, final Track track, final int i) {
176 
177         class Mdat implements Box {
178             ContainerBox parent;
179 
180             public ContainerBox getParent() {
181                 return parent;
182             }
183 
184             public void setParent(ContainerBox parent) {
185                 this.parent = parent;
186             }

188             public long getSize() {
189                 long size = 8; // I don't expect 2gig fragments
190                 for (ByteBuffer sample : getSamples(startSample, endSample, track, i)) {
191                     size += sample.limit();
192                 }
193                 return size;
194             }
195 
196             public String getType() {
197                 return "mdat";
198             }

200             public void getBox(WritableByteChannel writableByteChannel) throws IOException {
201                 List<ByteBuffer> bbs = getSamples(startSample, endSample, track, i);
202                 final List<ByteBuffer> samples = ByteBufferHelper.mergeAdjacentBuffers(bbs);
203                 ByteBuffer header = ByteBuffer.allocate(8);
204                 IsoTypeWriter.writeUInt32(header, l2i(getSize()));
205                 header.put(IsoFile.fourCCtoBytes(getType()));
206                 header.rewind();
207                 writableByteChannel.write(header);
208                 if (writableByteChannel instanceof GatheringByteChannel) {
209 
210                     int STEPSIZE = 1024;
211                     for (int i = 0; i < Math.ceil((double) samples.size() / STEPSIZE); i++) {
212                         List<ByteBuffer> sublist = samples.subList(
213                                 i * STEPSIZE, // start
214                                 (i + 1) * STEPSIZE < samples.size() ? (i + 1) * STEPSIZE : samples.size()); // end
215                         ByteBuffer sampleArray[] = sublist.toArray(new ByteBuffer[sublist.size()]);
216                         do {
217                             ((GatheringByteChannel) writableByteChannel).write(sampleArray);
218                         } while (sampleArray[sampleArray.length - 1].remaining() > 0);
219                     }
220                     //System.err.println(bytesWritten);
221                 } else {
222                     for (ByteBuffer sample : samples) {
223                         sample.rewind();
224                         writableByteChannel.write(sample);
225                     }
226                 }
227 
228             }
229 
230             public void parse(ReadableByteChannel readableByteChannel, ByteBuffer header, long contentSize, BoxParser boxParser) throws IOException {
231 
232             }
233         }
234 
235         return new Mdat();
236     }

278     protected List<? extends Box> createTruns(long startSample, long endSample, Track track, int sequenceNumber) {
279         List<ByteBuffer> samples = getSamples(startSample, endSample, track, sequenceNumber);
280 
281         long[] sampleSizes = new long[samples.size()];
282         for (int i = 0; i < sampleSizes.length; i++) {
283             sampleSizes[i] = samples.get(i).limit();
284         }
285         TrackRunBox trun = new TrackRunBox();
286 
287 
288         trun.setSampleDurationPresent(true);
289         trun.setSampleSizePresent(true);
290         List<TrackRunBox.Entry> entries = new ArrayList<TrackRunBox.Entry>(l2i(endSample - startSample));
291 
292 
293         Queue<TimeToSampleBox.Entry> timeQueue = new LinkedList<TimeToSampleBox.Entry>(track.getDecodingTimeEntries());
294         long left = startSample;
295         long curEntryLeft = timeQueue.peek().getCount();
296         long durationEntriesLeft = 0;
297         while (left > 0 && timeQueue.size() > 0) {
298             if (curEntryLeft >= left) {
299                 curEntryLeft -= left;
300                 left = 0;
301                 durationEntriesLeft = curEntryLeft;
302             } else {
303                 left -= curEntryLeft;
304                 timeQueue.poll();
305                 curEntryLeft = timeQueue.peek().getCount();
306             }
307         }
308 
309 
310         Queue<CompositionTimeToSample.Entry> compositionTimeQueue =
311                 track.getCompositionTimeEntries() != null && track.getCompositionTimeEntries().size() > 0 ?
312                         new LinkedList<CompositionTimeToSample.Entry>(track.getCompositionTimeEntries()) : null;
313         long compositionTimeEntriesLeft = compositionTimeQueue != null ? compositionTimeQueue.peek().getCount() : -1;
314 
315 
316 
317         trun.setSampleCompositionTimeOffsetPresent(compositionTimeEntriesLeft > 0);
318 
319         // fast forward composition stuff
320         for (long i = 1; i < startSample; i++) {
321             if (compositionTimeQueue != null) {
322                 trun.setSampleCompositionTimeOffsetPresent(true);
323                 if (--compositionTimeEntriesLeft == 0 && compositionTimeQueue.size() > 1) {
324                     compositionTimeQueue.remove();
325                     compositionTimeEntriesLeft = compositionTimeQueue.element().getCount();
326                 }
327             }
328         }
329 
330         boolean sampleFlagsRequired = (track.getSampleDependencies() != null && !track.getSampleDependencies().isEmpty() ||
331                 track.getSyncSamples() != null && track.getSyncSamples().length != 0);
332 
333         trun.setSampleFlagsPresent(sampleFlagsRequired);
334 
335         for (int i = 0; i < sampleSizes.length; i++) {
336             TrackRunBox.Entry entry = new TrackRunBox.Entry();
337             entry.setSampleSize(sampleSizes[i]);
338             if (sampleFlagsRequired) {
339                 //if (false) {
340                 SampleFlags sflags = new SampleFlags();
341 
342                 if (track.getSampleDependencies() != null && !track.getSampleDependencies().isEmpty()) {
343                     SampleDependencyTypeBox.Entry e = track.getSampleDependencies().get(i);
344                     sflags.setSampleDependsOn(e.getSampleDependsOn());
345                     sflags.setSampleIsDependedOn(e.getSampleIsDependentOn());
346                     sflags.setSampleHasRedundancy(e.getSampleHasRedundancy());
347                 }
348                 if (track.getSyncSamples() != null && track.getSyncSamples().length > 0) {
349                     // we have to mark non-sync samples!
350                     if (Arrays.binarySearch(track.getSyncSamples(), startSample + i) >= 0) {
351                         sflags.setSampleIsDifferenceSample(false);
352                         sflags.setSampleDependsOn(2);
353                     } else {
354                         sflags.setSampleIsDifferenceSample(true);
355                         sflags.setSampleDependsOn(1);
356                     }
357                 }
358                 // i don't have sample degradation
359                 entry.setSampleFlags(sflags);
360 
361             }
362 
363             entry.setSampleDuration(timeQueue.peek().getDelta());
364             if (--durationEntriesLeft == 0 && timeQueue.size() > 1) {
365                 timeQueue.remove();
366                 durationEntriesLeft = timeQueue.peek().getCount();
367             }
368 
369             if (compositionTimeQueue != null) {
370                 trun.setSampleCompositionTimeOffsetPresent(true);
371                 entry.setSampleCompositionTimeOffset(compositionTimeQueue.peek().getOffset());
372                 if (--compositionTimeEntriesLeft == 0 && compositionTimeQueue.size() > 1) {
373                     compositionTimeQueue.remove();
374                     compositionTimeEntriesLeft = compositionTimeQueue.element().getCount();
375                 }
376             }
377             entries.add(entry);
378         }
379 
380         trun.setEntries(entries);
381 
382         return Collections.singletonList(trun);
383     }

399     protected Box createMvhd(Movie movie) {
400         MovieHeaderBox mvhd = new MovieHeaderBox();
401         mvhd.setVersion(1);
402         mvhd.setCreationTime(DateHelper.convert(new Date()));
403         mvhd.setModificationTime(DateHelper.convert(new Date()));
404         long movieTimeScale = movie.getTimescale();
405         long duration = 0;
406 
407         for (Track track : movie.getTracks()) {
408             long tracksDuration = getDuration(track) * movieTimeScale / track.getTrackMetaData().getTimescale();
409             if (tracksDuration > duration) {
410                 duration = tracksDuration;
411             }
412 
413 
414         }
415 
416         mvhd.setDuration(duration);
417         mvhd.setTimescale(movieTimeScale);
418         // find the next available trackId
419         long nextTrackId = 0;
420         for (Track track : movie.getTracks()) {
421             nextTrackId = nextTrackId < track.getTrackMetaData().getTrackId() ? track.getTrackMetaData().getTrackId() : nextTrackId;
422         }
423         mvhd.setNextTrackId(++nextTrackId);
424         return mvhd;
425     }

441     protected Box createTfra(Track track, IsoFile isoFile) {
442         TrackFragmentRandomAccessBox tfra = new TrackFragmentRandomAccessBox();
443         tfra.setVersion(1); // use long offsets and times
444         List<TrackFragmentRandomAccessBox.Entry> offset2timeEntries = new LinkedList<TrackFragmentRandomAccessBox.Entry>();
445         List<Box> boxes = isoFile.getBoxes();
446         long offset = 0;
447         long duration = 0;
448         for (Box box : boxes) {
449             if (box instanceof MovieFragmentBox) {
450                 List<TrackFragmentBox> trafs = ((MovieFragmentBox) box).getBoxes(TrackFragmentBox.class);
451                 for (int i = 0; i < trafs.size(); i++) {
452                     TrackFragmentBox traf = trafs.get(i);
453                     if (traf.getTrackFragmentHeaderBox().getTrackId() == track.getTrackMetaData().getTrackId()) {
454                         // here we are at the offset required for the current entry.
455                         List<TrackRunBox> truns = traf.getBoxes(TrackRunBox.class);
456                         for (int j = 0; j < truns.size(); j++) {
457                             List<TrackFragmentRandomAccessBox.Entry> offset2timeEntriesThisTrun = new LinkedList<TrackFragmentRandomAccessBox.Entry>();
458                             TrackRunBox trun = truns.get(j);
459                             for (int k = 0; k < trun.getEntries().size(); k++) {
460                                 TrackRunBox.Entry trunEntry = trun.getEntries().get(k);
461                                 SampleFlags sf = null;
462                                 if (k == 0 && trun.isFirstSampleFlagsPresent()) {
463                                     sf = trun.getFirstSampleFlags();
464                                 } else if (trun.isSampleFlagsPresent()) {
465                                     sf = trunEntry.getSampleFlags();
466                                 } else {
467                                     List<MovieExtendsBox> mvexs = isoFile.getMovieBox().getBoxes(MovieExtendsBox.class);
468                                     for (MovieExtendsBox mvex : mvexs) {
469                                         List<TrackExtendsBox> trexs = mvex.getBoxes(TrackExtendsBox.class);
470                                         for (TrackExtendsBox trex : trexs) {
471                                             if (trex.getTrackId() == track.getTrackMetaData().getTrackId()) {
472                                                 sf = trex.getDefaultSampleFlags();
473                                             }
474                                         }
475                                     }
476 
477                                 }
478                                 if (sf == null) {
479                                     throw new RuntimeException("Could not find any SampleFlags to indicate random access or not");
480                                 }
481                                 if (sf.getSampleDependsOn() == 2) {
482                                     offset2timeEntriesThisTrun.add(new TrackFragmentRandomAccessBox.Entry(
483                                             duration,
484                                             offset,
485                                             i + 1, j + 1, k + 1));
486                                 }
487                                 duration += trunEntry.getSampleDuration();
488                             }
489                             if (offset2timeEntriesThisTrun.size() == trun.getEntries().size() && trun.getEntries().size() > 0) {
490                                 // Oooops every sample seems to be random access sample
491                                 // is this an audio track? I don't care.
492                                 // I just use the first for trun sample for tfra random access
493                                 offset2timeEntries.add(offset2timeEntriesThisTrun.get(0));
494                             } else {
495                                 offset2timeEntries.addAll(offset2timeEntriesThisTrun);
496                             }
497                         }
498                     }
499                 }
500             }
501 
502 
503             offset += box.getSize();
504         }
505         tfra.setEntries(offset2timeEntries);
506         tfra.setTrackId(track.getTrackMetaData().getTrackId());
507         return tfra;
508     }

549     protected Box createTkhd(Movie movie, Track track) {
550         TrackHeaderBox tkhd = new TrackHeaderBox();
551         tkhd.setVersion(1);
552         int flags = 0;
553         if (track.isEnabled()) {
554             flags += 1;
555         }
556 
557         if (track.isInMovie()) {
558             flags += 2;
559         }
560 
561         if (track.isInPreview()) {
562             flags += 4;
563         }
564 
565         if (track.isInPoster()) {
566             flags += 8;
567         }
568         tkhd.setFlags(flags);
569 
570         tkhd.setAlternateGroup(track.getTrackMetaData().getGroup());
571         tkhd.setCreationTime(DateHelper.convert(track.getTrackMetaData().getCreationTime()));
572         // We need to take edit list box into account in trackheader duration
573         // but as long as I don't support edit list boxes it is sufficient to
574         // just translate media duration to movie timescale
575         tkhd.setDuration(getDuration(track) * movie.getTimescale() / track.getTrackMetaData().getTimescale());
576         tkhd.setHeight(track.getTrackMetaData().getHeight());
577         tkhd.setWidth(track.getTrackMetaData().getWidth());
578         tkhd.setLayer(track.getTrackMetaData().getLayer());
579         tkhd.setModificationTime(DateHelper.convert(new Date()));
580         tkhd.setTrackId(track.getTrackMetaData().getTrackId());
581         tkhd.setVolume(track.getTrackMetaData().getVolume());
582         return tkhd;
583     }
